            if ($("#list-header").is(':visible')) {
                app.filterCount();
            } else {
                app.filter();
            }
        });

        // 検索フォームの一覧表示ボタン
        $("#search-condition .list-btn").click(function() {
            $(this).button("loading");
            app.updateList(null, function() {
                app.chargeSearchResultBtns();
                app.hideSearchCondition();
            });
        });

        // 検索結果モーダルの一覧表示ボタン
        $("#search-result-modal .list-btn").click(function() {
            $(this).button("loading");
            app.filter(function() {
                app.updateList(null, function() {
                    $("#search-result-modal .list-btn").button("reset");
                    $("#search-result-modal").modal('hide');
                });
            });
        });

        // 検索条件に戻るボタン
        $(".search-condition-btn").click(app.showSearchCondition);

        // 検索フォームの絞込条件ボタン
        $("#search-condition .filter-condition-btn").click(function() {
            app.chargeSearchResultBtns();
            app.showFilterCondition();
        });

        // 検索結果一覧の絞込条件ボタン
        $("#list-header .filter-condition-btn").click(app.showFilterCondition);

        $("#search-result-modal .filter-condition-btn").click(function() {

        $(".detail-tabs a").click(function() {

    // 条件クリア
    app.clearCondition = function() {
        var currentConditionId = $("#saved-condition").val();
        if (currentConditionId !== '') {
            app.setSavedConditionToFormById(currentConditionId);
        } else {
            app.initExtraCondition()
            app.setSearchFilterLabel();
            $("textarea[name=name]").val("");
            $("textarea[name=similar_code]").val("");
            $("textarea[name=international_class]").val("");
        }
        $(".search-result-item").hide();
        app.clearError();
    };

    // 検索実行
    app.search = function() {
        if (app.vars.waitingResponse) return;

        var params = $("#form").serialize();
        params += '&search_id=' + app.vars.searchId;

        $(".search-result-item").hide();
        $("#search-btn").button("loading");
        app.vars.waitingResponse = true;
        app.clearError();

        $.ajax({
            type: 'POST',
            url: '/ruijistd/search',
            data: params,
            dataType: 'json'
        }).done(function(data) {
            if (data.status == 'ok') {
                app.vars.searchResultCount = data.values.count;
                var message = Number(data.values.count).toLocaleString() + "件の情報が見つかりました";
                if (data.values.over) {
                    message = "最大件数を超える情報が見つかりました<br>10,000件までの情報を表示できます"
                }
                $(".search-result-message").html(message);
                $(".search-result-item").show();
                app.resetSearchResultBtns();
                app.setFilterCondition();
            } else {
                app.showError(data.errors);
            }
        }).always(function() {
            $("#search-btn").button("reset");
            app.vars.waitingResponse = false;
        });
    };

    // 絞込実行
    app.filter = function(callback) {
        if (app.vars.waitingResponse) return;

        $("#filter-btn").button("loading");
        app.vars.waitingResponse = true;
        app.clearError();

        $.ajax({
            type: 'POST',
            url: '/ruijistd/filter',
            data: app.getFilterParams(),
            dataType: 'json'
        }).done(function(data) {
            if (data.status == 'ok') {
                $("#filter-btn").button("reset");
                if (callback && typeof(callback) == 'function') {
                    callback();
                } else {
                    var message = Number(data.values.count).toLocaleString() + "件 / " + Number(app.vars.searchResultCount).toLocaleString() + "件の情報が見つかりました";
                    $("#similar-filter-modal").modal('hide');
                    $(".search-result-message").html(message);
                }
            } else {
                app.showError(data.errors);
            }
        }).always(function() {
            app.vars.waitingResponse = false;
        });
    };

    // 絞込件数取得
    app.filterCount = function() {
        if (app.vars.waitingResponse) return;

        $("#filter-btn").button("loading");
        app.vars.waitingResponse = true;
        app.clearError();

        $.ajax({
            type: 'POST',
            url: '/ruijistd/filtercount',
            data: app.getFilterParams(),
            dataType: 'json'
        }).done(function(data) {
            if (data.status == 'ok') {
                var message = Number(data.values.count).toLocaleString() + "件 / " + Number(app.vars.searchResultCount).toLocaleString() + "件の情報が見つかりました";
                $("#filter-btn").button("reset");
                $("#similar-filter-modal").modal('hide');
                $(".search-result-message").html(message);
                $("#search-result-modal").modal();
            } else {
                app.showError(data.errors);
            }
        }).always(function() {
            app.vars.waitingResponse = false;
        });
    };

    // 絞込条件内容をセット
    app.setFilterCondition = function() {
        if (!app.vars.filterTemplates) {
            app.vars.filterTemplates = {};
        }

        var params = 'search_id=' + app.vars.searchId;

        $.get('/ruijistd/filtercondition', params, function(data) {
            if (data.status == 'ok') {
                app.setFilterConditionTable(data.values, 'shoko');
            }
        }, 'json');
    };

    // 絞込条件表示時に呼び出す
    app.requestShowFilter = function() {
        var params = 'search_id=' + app.vars.searchId;

        $.get('/ruijistd/showfilter', params, function(data) {
            app.updateChargeNums();
        }, 'json');
    };

    // 絞込条件表示
    app.showFilterCondition = function() {
        app.updateFilterListDisplay('shoko');
        $("#similar-filter-modal").modal();
        app.requestShowFilter();
    };

    app.updateList = function(page, callback) {
        var url = app.getUpdateListUrl(page);

        if ($('#list').is(':visible')) {
            app.showLoading();
        }

        $("#list").load(url, function() {
            app.setListAndDetailLayout();
            app.setListHeaderColAction();
            app.setListRowHelpAction();
            app.setListRowDetailBtnClickEvent();
            app.scrollToSelectedListNo();
            app.updateChargeNums();

            if ($('#loading-modal').is(':visible')) {
                app.hideLoading();
            }

            if (callback && typeof(callback) == 'function') {
                callback();
            }
        });
    };

    app.getConditionParams = function() {
        var params = 'search_id=' + app.vars.searchId;
        params += '&' + $("#form").serialize();
        return params;
    };

    app.setConditionToForm = function(condition) {
        var typeMap = {
            '103': 'name',
            '106': 'similar_code',
            '104': 'international_class'
        };
        $.each(condition.condition_desc_array, function(i, item) {
            var name = typeMap[item.type];
            $("textarea[name=" + name + "]").val(item.str);
        });

        // 追加条件をセット
        $("input[name=sonzoku_kind]").val([condition.extra_condition_map.sonzoku_kind]);
        $("input[name=sikko_kind]").val([condition.extra_condition_map.sikko_kind]);
        $("input[name=shukya_kind]").val([condition.extra_condition_map.shukya_kind]);
        $("input[name=cb_1]").val([condition.extra_condition_map.reg_kind]);
        app.setSearchFilterLabel();
    };
</script>
