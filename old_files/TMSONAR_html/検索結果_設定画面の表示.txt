    app.getConditionParams = function() {
        // 条件パラメータ名
        var paramNames = ['left_braces', 'types', 'strs', 'inner_logicalsigns', 'right_braces', 'outer_logicalsigns'];

        // 送信するパラメータを作成
        var params = 'search_id=' + app.vars.searchId;
        params += '&' + $("#header-condition-form").serialize();
        params += '&num_rows[normal]=' + $("#normal-panel .strs").length;
        params += '&num_rows[not]=' + $("#not-panel .strs").length;
        paramNames.forEach(function(name) {
            $("#normal-panel ." + name).each(function(index, element) {
                params += '&' + name + '[normal][' + index + ']=';
                if ($(element).val() !== null) {
                    params += encodeURIComponent($(element).val());
                }
            });
            $("#not-panel ." + name).each(function(index, element) {
                params += '&' + name + '[not][' + index + ']=';
                if ($(element).val() !== null) {
                    params += encodeURIComponent($(element).val());
                }
            });
        });

        return params;
    };

    app.setConditionToForm = function(condition) {
        // 抽出条件、除外条件をクリア
        $("#normal-panel tbody").empty();
        $("#not-panel tbody").empty();

        if (condition) {
            // 条件行をセット
            $.each(condition.condition_desc_array, function (i, item) {
                var panelId = 'normal-panel';
                if (item.mode != 'normal') {
                    panelId = 'not-panel';
                }
                $("#" + panelId + " tbody").append(app.vars.conditionRowTemplate);
                $("#" + panelId + " tbody .types:last").val(item.type);
                $("#" + panelId + " tbody .strs:last").val(item.str);
                $("#" + panelId + " tbody .left_braces:last").val(item.left_brace);
                $("#" + panelId + " tbody .right_braces:last").val(item.right_brace);
                $("#" + panelId + " tbody .inner_logicalsigns:last").val(item.inner_logicalsign);
                if (item.outer_logicalsign > 0) {
                    $("#" + panelId + " tbody .outer_logicalsigns:last").val(item.outer_logicalsign);
                }
            });

            // 追加条件をセット
            $("input[name=sonzoku_kind]").val([condition.extra_condition_map.sonzoku_kind]);
            $("input[name=sikko_kind]").val([condition.extra_condition_map.sikko_kind]);
            $("input[name=shukya_kind]").val([condition.extra_condition_map.shukya_kind]);
            app.setSearchFilterLabel();
            $("input[name=cb_1]").val([condition.extra_condition_map.reg_kind]);
            $("input[name=cb_2]").val([condition.extra_condition_map.data_kind]);
        }

        // 抽出条件は最低8行表示
        var rows = $("#normal-panel tbody tr");
        for (var i = rows.length; i < 8; i++) {
            $("#normal-panel tbody").append(app.vars.conditionRowTemplate);
        }

        // 除外条件は最低1行表示
        var rows = $("#not-panel tbody tr");
        for (var i = rows.length; i < 1; i++) {
            $("#not-panel tbody").append(app.vars.conditionRowTemplate);
        }

        app.setAllConditionRowStatus();
    };

    app.hideSearchResultItem = function() {
        $(".search-result-item").hide();
        $("#pickup-list-btn").hide();
    };

    // 条件クリア
    app.clearCondition = function() {
        var currentConditionId = $("#saved-condition").val();
        if (currentConditionId !== '') {
            app.setSavedConditionToFormById(currentConditionId);
        } else {
            app.initExtraCondition();
            app.setSearchFilterLabel();
            app.setConditionToForm();
        }
        app.hideSearchResultItem();
        app.clearError();
    };

    // 検索実行
    app.search = function() {
        if (app.vars.waitingResponse) return;

        var params = app.getConditionParams();

        app.hideSearchResultItem();
        $("#search-btn").button("loading");
        app.vars.waitingResponse = true;
        app.clearError();

        $.ajax({
            type: 'POST',
            url: '/' + app.serviceName + '/search',
            data: params,
            dataType: 'json'
        }).done(function(data) {
            if (data.status == 'ok') {
                var message = Number(data.values.count).toLocaleString() + "件の情報が見つかりました";
                if (data.values.over) {
                    message = "最大件数を超える情報が見つかりました<br>10,000件までの情報を表示できます"
                }
                $("#list-btn").attr("data-use-pickup", 0);
                if (data.values.is_available_pickup) {
                    $("#pickup-list-btn").show();
                } else {
                    $("#pickup-list-btn").hide();
                }
                app.resetSearchResultBtns();
                $(".search-result-message").html(message);
                $(".search-result-item").show();
                $("#list-btn").attr("data-page", "");
            } else {
                app.showError(data.errors);
            }
        }).always(function() {
            $("#search-btn").button("reset");
            app.vars.waitingResponse = false;
        });
    };

    app.updateList = function(page, usePickup, callback) {
        var url = app.getUpdateListUrl(page);

        if (usePickup) {
            url += '&use_pickup=1';
        }

        if ($('#list').is(':visible')) {
            app.showLoading();
        }

        $("#list").load(url, function() {
            app.setListAndDetailLayout();
            app.setListHeaderColAction();
            app.setListRowHelpAction();
            app.setListRowDetailBtnClickEvent();
            app.scrollToSelectedListNo();
            app.updateChargeNums();

            if ($('#loading-modal').is(':visible')) {
                app.hideLoading();
            }

            if (callback && typeof(callback) == 'function') {
                callback();
            }
        });
    };

    app.isConditionRowHasHelpRow = function(row) {
        return row.next().hasClass("help-type");
    };
    
    app.addHelpRow = function(row) {
        if (app.isConditionRowHasHelpRow(row)) {
            return false;
        }
        
        var type = row.find(".types").val();
        if (type) {
            var helpHtml = $("#help-type-" + type).html();
            if (helpHtml) {
                var helpRowHtml = '<tr class="help-type help-type-' + type + '" style="display: none;"><td></td><td colspan="7">' + helpHtml + '</td></tr>';
                row.after(helpRowHtml);
                return true;
            }
        }

        return false;
    };
    
    app.removeHelpRow = function(row) {
        if (!app.isConditionRowHasHelpRow(row)) {
            return;
        }
        
        var helpRow = row.next();
        helpRow.remove();
    };

    app.setTypeHelpIconDisplay = function(row) {
        if (row.find(".types").val()) {
            row.find(".type-help-icon").show();
        } else {
            row.find(".type-help-icon").hide();
        }
    };
    
    app.toggleTypeHelp = function(obj) {
        var row = $(obj).closest("tr");
        var existsHelpRow = true;
        if (!app.isConditionRowHasHelpRow(row)) {
            existsHelpRow = app.addHelpRow(row);
        }
        if (existsHelpRow) {
            var helpRow = row.next();
            helpRow.toggle();
        }
    };

    app.addConditionRow = function(obj) {
        var row = $(obj).closest("tr");
        var newRowHtml = "<tr>" + $(obj).closest("tr").html() + "</tr>";
        if (app.isConditionRowHasHelpRow(row)) {
            row = row.next();
        }
        row.after(newRowHtml);
        var addedRow = row.next();
        addedRow.find(".dictionary-assistance-btn").remove();
        app.setTypeHelpIconDisplay(addedRow);
        app.setOuterLogicalSignStatus();
    };

    app.removeConditionRow = function(obj) {
        if ($(obj).closest("tbody").find("tr").length == 1) {
            // 1行しかない場合は削除しない
            return;
        }

        var row = $(obj).closest("tr");
        if (app.isConditionRowHasHelpRow(row)) {
            app.removeHelpRow(row);
        }
        row.remove();
        app.setOuterLogicalSignStatus();
    };

    app.setConditionRowStatusByTypes = function(obj) {
        var row = $(obj).closest("tr");
        var type = row.find(".types option:selected");
        var dictionaryAssistanceBtnHtml = '<button type="button" class="btn btn-default dictionary-assistance-btn"></button>';
        row.find(".dictionary-assistance-btn").remove();
        if (type.attr('data-dictionary') && (app.vars.isDictionaryEnabled || app.isTypeViennaCode(type))) {
            row.find(".btn-group").append(dictionaryAssistanceBtnHtml);
            var btnName = "辞書";
            if (app.isTypeViennaCode(type)) {
                btnName = "補助";
            }
            row.find(".dictionary-assistance-btn").html(btnName).click(app.showDictionaryModal);
        } else if (type.attr('data-assistance')) {
            row.find(".btn-group").append(dictionaryAssistanceBtnHtml);
            row.find(".dictionary-assistance-btn").show().html("補助").click(app.showAssistanceModal);
        }
        app.removeHelpRow(row);
        app.setTypeHelpIconDisplay(row);
    };

    app.showAssistanceModal = function() {
        if ($(this).prop("disabled")) return;

        var row = $(this).closest("tr");
        row.addClass("assistance-row");

        var type = row.find(".types").val();

        $("#assistance-modal .modal-content").load("/" + app.serviceName + "/assistance/type/" + type, function() {
            if (type == '109') {
                // 審決分類
                app.setShinketsuAssistanceContent();
                app.setShinketsuAssistanceText();
                $("#shinketsu-assistance-shinpan .radio-row").click(app.setShinketsuAssistanceKetsuronTabDisplay);
            }
            $("#assistance-modal").modal();
            $("#assistance-modal .bulk-select-row").click(function() {
                var values = $(this).data("values").split(",");
                var rows = $("#assistance-modal .selectable-row");
                for (var i = 0; i < rows.length; i++) {
                    var row = $(rows[i]);
                    var value = row.data("value");
                    app.resetRow(row);
                    if (values.includes(String(value))) {
                        app.selectRow(row);
                    }
                }
            });

            $("#assistance-modal .selectable-row").click(function() {
                app.selectRow($(this));
            });
            $("#assistance-modal .radio-row").click(function() {
                app.selectRadioRow($(this));
            });
            $("#assistance-modal .selectable-row").click(app.setShinketsuAssistanceText);
            $("#assistance-modal .radio-row").click(app.setShinketsuAssistanceText);
        });
    };

    app.hideAssistanceModal = function() {
        $(".assistance-row").removeClass("assistance-row");
        $("#assistance-modal").modal("hide");
    };

    app.showDictionaryModal = function() {
        if ($(this).prop("disabled")) return;

        $(".dictionary-input").removeClass("dictionary-input");
        var row = $(this).closest("tr");
        row.find(".strs").addClass("dictionary-input");

        var type = row.find(".types").val();

        // （英語）指定商品／役務
        if (type == 140) {
            type = 107;
        }

        $("#dictionary-modal .modal-content").load("/dictionary/modal" + type, params, function() {
            if (type == 112) {
                // ウィーンコード
                $("#dictionary-modal .modal-dialog").width(1200);
            }
            $("#dictionary-modal").modal();
        });
    };

    app.hideDictionaryModal = function() {
        $("#dictionary-modal").modal('hide');
    };

    app.searchDictionary = function(type) {
        if (app.vars.waitingResponse) return;

        app.vars.waitingResponse = true;
        $(".dictionary-search-btn").button("loading");

        var params = 'service_name=' + app.serviceName +
            '&search_id=' + app.vars.searchId +
            '&' + $("#dictionary-form").serialize();

        $.post('/dictionary/search' + type, params, function(data) {
            if (data.status == 'ok') {
                $("#dictionary-search-result").html(data.html);
                $(".dictionary-table tbody tr").click(function() {
                    if ($(this).hasClass("info")) {
                        $(this).removeClass("info");
                        $(this).find("input").prop("checked", false);
                    } else {
                        $(this).addClass("info");
                        $(this).find("input").prop("checked", true);
                    }
                });
            } else {
                app.alert(data.errors);
            }
            app.updateChargeNums();
            app.vars.waitingResponse = false;
            $(".dictionary-search-btn").button("reset");
        }, 'json');
    };
